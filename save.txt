git init                      "Inicializa o repositório Git local"
git add .                    "Adiciona todos os arquivos para o controle de versão"
git commit -m "primeiro commit"   "Salva as alterações com uma mensagem"
git branch -M main           "Renomeia a branch principal para 'main'"
git remote add origin https://github.com/usuario/repositorio.git   "Liga seu repositório local ao do GitHub"
git pull origin main --allow-unrelated-histories   "Sincroniza com o repositório remoto (caso ele já tenha arquivos)"
git push -u origin main      "Envia os arquivos locais para o GitHub e vincula a branch"
